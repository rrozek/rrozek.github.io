    Appendix D: Acknowledgment (Ack) Schema

    Acknowledgment messages form the backbone of signal delivery confirmation. They are sent automatically in response to a valid, decryptable, and authenticated incoming message.

    Acks serve to:

    • Confirm delivery and routing success
    • Prevent retries for confirmed messages
    • Aid clients in signaling liveliness
    • Provide metadata for latency and diagnostics

    Optional extensions to this schema may include:

    • Signature of the original message hash for integrity confirmation
    • Forwarding chain identifier (in store-and-forward flows)
    • Relayed timestamp (from intermediary node)

    For sensitive messages (e.g., credential issuance or payments), the status may elevate to verified only after backend or authority validation.

    Acknowledgments notify the sender that their signal was received, decrypted, and accepted for processing. This schema supports delivery monitoring and prevents retry loops.

    Base Acknowledgment Structure

    interface BaseAcknowledgment {
  id: string;
  type: 'ack';
  version: string;
  timestamp: string; // RFC3339
  originalMessageId: string;
  sender: string; // sfname of ack sender
  recipient: string; // sfname of original sender
  status: 'delivered' | 'verified' | 'pending' | 'failed';
}

    Acknowledgment Processing Flow

    Sender-Side Processing

    class AcknowledgmentProcessor {
  private pendingAcknowledgments = new Map<string, AcknowledgmentPromise>();

  async sendMessage(message: SignalMessage): Promise<DeliveryResult> {
    const deliveryPromise = new AcknowledgmentPromise(message.id);
    this.pendingAcknowledgments.set(message.id, deliveryPromise);

    // Send message
    await this.signalTransport.send(message);

    // Set timeout for acknowledgment
    const timeoutPromise = new Promise((_, reject) => {
      setTimeout(() => reject(new Error('Acknowledgment timeout')), ACK_TIMEOUT_MS);
    });

    try {
      const acknowledgment = await Promise.race([
        deliveryPromise,
        timeoutPromise
      ]);

      return this.processAcknowledgment(acknowledgment);
    } catch (error) {
      // Handle timeout or other errors
      return this.handleAcknowledgmentFailure(message, error);
    }
  }

  processAcknowledgment(ack: Acknowledgment): DeliveryResult {
    switch (ack.status) {
      case 'delivered':
        return { status: 'delivered', latency: this.calculateLatency(ack) };
      case 'verified':
        return { status: 'verified', latency: this.calculateLatency(ack) };
      case 'failed':
        return { status: 'failed', error: ack.error_details };
      case 'pending':
        return { status: 'pending', estimatedCompletion: ack.processing_metadata?.estimated_completion_time };
      default:
        return { status: 'unknown' };
    }
  }
}

    Recipient-Side Generation

    async function generateAcknowledgment(
  originalMessage: SignalMessage,
  processingResult: ProcessingResult
): Promise<Acknowledgment> {
  const baseAck: BaseAcknowledgment = {
    id: generateAcknowledgmentId(),
    type: 'ack',
    version: '1.0',
    timestamp: new Date().toISOString(),
    original_message_id: originalMessage.id,
    sender: getCurrentIdentity().sfname,
    recipient: originalMessage.sender,
    status: processingResult.status
  };

  // Add processing metadata
  const processingMetadata = {
    received_at: processingResult.receivedAt,
    decrypted_at: processingResult.decryptedAt,
    processed_at: processingResult.processedAt,
    processing_time_ms: processingResult.processingTime,
    routing_strategy_used: processingResult.routingStrategy,
    hops_taken: processingResult.hopsTaken
  };

  // Add validation results
  const validationResults = {
    signature_verified: processingResult.signatureValid,
    decryption_successful: processingResult.decryptionSuccessful,
    content_integrity: processingResult.integrityCheck,
    replay_check: processingResult.replayCheck
  };

  // Add delivery confirmation
  const deliveryConfirmation = {
    message_delivered_to_application: processingResult.deliveredToApp,
    application_response: processingResult.appResponse,
    notification_sent: processingResult.notificationSent
  };

  // Sign the acknowledgment
  const ackPayload = { ...baseAck, processingMetadata, validationResults, deliveryConfirmation };
  const signature = await signAcknowledgment(ackPayload);

  return {
    ...ackPayload,
    signature
  };
}

    Acknowledgment Validation

    Signature Verification

    async function verifyAcknowledgmentSignature(
  acknowledgment: Acknowledgment,
  senderPublicKey: Uint8Array
): Promise<boolean> {
  const { signature, ...ackWithoutSignature } = acknowledgment;

  // Canonicalize the acknowledgment payload
  const canonicalized = canonicalizeJson(ackWithoutSignature);

  // Verify signature
  return await ed25519Verify(
    canonicalized,
    base64Decode(signature),
    senderPublicKey
  );
}

    Content Validation

    function validateAcknowledgmentContent(acknowledgment: Acknowledgment): boolean {
  // Validate required fields
  const requiredFields = ['id', 'type', 'timestamp', 'original_message_id', 'status'];
  const hasRequiredFields = requiredFields.every(field => field in acknowledgment);

  if (!hasRequiredFields) {
    return false;
  }

  // Validate timestamp
  if (!isValidRFC3339(acknowledgment.timestamp)) {
    return false;
  }

  // Validate status
  const validStatuses = ['delivered', 'verified', 'pending', 'failed'];
  if (!validStatuses.includes(acknowledgment.status)) {
    return false;
  }

  // Validate processing metadata if present
  if (acknowledgment.processing_metadata) {
    if (!validateProcessingMetadata(acknowledgment.processing_metadata)) {
      return false;
    }
  }

  return true;
}

    Retry Logic Integration

    Acknowledgments prevent unnecessary retries and provide feedback for optimization:

    class RetryManager {
  private retryQueue = new Map<string, RetryItem>();

  handleAcknowledgment(acknowledgment: Acknowledgment): void {
    const originalMessageId = acknowledgment.original_message_id;

    if (acknowledgment.status === 'delivered' || acknowledgment.status === 'verified') {
      // Remove from retry queue
      this.retryQueue.delete(originalMessageId);

      // Update delivery statistics
      this.updateDeliveryStats(acknowledgment);
    } else if (acknowledgment.status === 'failed') {
      // Handle failed acknowledgment
      this.handleFailedAcknowledgment(acknowledgment);
    }
  }

  private handleFailedAcknowledgment(acknowledgment: Acknowledgment): void {
    const retryItem = this.retryQueue.get(acknowledgment.original_message_id);

    if (retryItem && acknowledgment.error_details?.retryable) {
      // Schedule retry with backoff
      this.scheduleRetry(retryItem);
    } else {
      // Permanent failure
      this.markAsFailed(acknowledgment);
    }
  }
}

    Monitoring and Analytics

    Delivery Metrics

    interface DeliveryMetrics {
  totalMessages: number;
  deliveredMessages: number;
  verifiedMessages: number;
  failedMessages: number;
  averageLatency: number;
  deliveryRate: number;
  retryRate: number;
}

class DeliveryAnalytics {
  private metrics: DeliveryMetrics = {
    totalMessages: 0,
    deliveredMessages: 0,
    verifiedMessages: 0,
    failedMessages: 0,
    averageLatency: 0,
    deliveryRate: 0,
    retryRate: 0
  };

  recordAcknowledgment(acknowledgment: Acknowledgment): void {
    this.metrics.totalMessages++;

    switch (acknowledgment.status) {
      case 'delivered':
        this.metrics.deliveredMessages++;
        break;
      case 'verified':
        this.metrics.verifiedMessages++;
        this.metrics.deliveredMessages++;
        break;
      case 'failed':
        this.metrics.failedMessages++;
        break;
    }

    // Update latency calculations
    if (acknowledgment.processing_metadata?.processing_time_ms) {
      this.updateAverageLatency(acknowledgment.processing_metadata.processing_time_ms);
    }

    // Update rates
    this.updateRates();
  }
}
