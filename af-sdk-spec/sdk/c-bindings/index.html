    C API Bindings

    C language bindings for the MatterFi SDK.

    Data Structures

    // Unified error codes for all API operations
typedef enum {
    MATTERFI_SUCCESS = 0,
    MATTERFI_INTERNAL_ERROR = 1,
    MATTERFI_RECIPIENT_INVALID = 2,
    MATTERFI_NOT_FOUND = 4,
    MATTERFI_ADDRESS_CALCULATING_ERROR = 5,
    MATTERFI_ADDRESS_CONVERSION_ERROR = 6,
    MATTERFI_INVALID_CLAIM = 7,
    MATTERFI_INVALID_ACCOUNT = 8,
    MATTERFI_INVALID_SEED = 9,
    MATTERFI_INVALID_RECIPIENT = 10,
    MATTERFI_INVALID_SFNAME = 11,
    MATTERFI_SFNAME_IN_USE = 12,
    MATTERFI_INSUFFICIENT_FUNDS = 13,
    MATTERFI_FEE_TOO_LOW = 14,
    MATTERFI_TRANSFER_FAILED = 15,
    MATTERFI_PAYMENT_NOT_FOUND = 16,
    MATTERFI_PAYMENT_CANNOT_CANCEL = 17,
    MATTERFI_INVALID_TOKEN = 18,
    MATTERFI_STAKE_TOO_LOW = 19,
    MATTERFI_UNSTAKE_NOT_ALLOWED = 20,
    MATTERFI_VALIDATOR_NOT_FOUND = 21
} matterfi_error_t;

// Unified result structure for all API responses
typedef struct {
    matterfi_error_t error;
    char* data;  // Generic data field (string, JSON, etc.) - ownership transferred
} matterfi_result_t;

typedef enum {
    MATTERFI_ETH,
    MATTERFI_SOL
} matterfi_network_t;

    Chain ID Standards

    The SDK uses SLIP-44 standard coin types for blockchain identification:

    // SLIP-44 Coin Types (https://github.com/satoshilabs/slips/blob/master/slip-0044.md)
#define MATTERFI_CHAIN_BITCOIN     0    // Bitcoin
#define MATTERFI_CHAIN_ETHEREUM    60   // Ethereum
#define MATTERFI_CHAIN_TESTNET     1    // Testnets (all networks)

    // Initialize SDK with configuration

    Seed Management

    // Generate new seed
// Returns: matterfi_result_t with data containing space-separated mnemonic words
matterfi_result_t matterfi_generate_seed();

// Import existing seed
// input_words: Space-separated mnemonic words
// password: Optional passphrase
// seed_style: "bip39" (enum value)
// language: "en" (enum value)
// Returns: matterfi_result_t with error indicating success/failure
matterfi_result_t matterfi_import_seed(
    const char* input_words,
    const char* password,
    int seed_style,
    int language
);

// Validate seed
// seed: Seed to validate
// password: Optional passphrase
// Returns: matterfi_result_t with data containing "true"/"false"
matterfi_result_t matterfi_validate_seed(const char* seed, const char* password);

// Get recovery words from seed
// Returns: matterfi_result_t with data containing space-separated words
matterfi_result_t matterfi_get_recovery_words();

    Nym (Identity) Management

    // Create nym from seed
// seed: Previously imported seed
// password: Password for cryptographic operations
// alias: Human-readable name for the nym
// Returns: matterfi_result_t with error indicating success/failure
matterfi_result_t matterfi_create_nym(
    const char* seed,
    const char* password,
    const char* alias
);

// Get nym public key
// Returns: matterfi_result_t with data containing hex-encoded public key
matterfi_result_t matterfi_get_nym_public_key();

// Get nym ID
// Returns: matterfi_result_t with data containing base58-encoded nym ID
matterfi_result_t matterfi_get_nym_id();

// Add claim to nym
// claim_data: JSON string with claim information
// Returns: matterfi_result_t with error indicating success/failure
matterfi_result_t matterfi_add_claim_to_nym(const char* claim_data);

    Error Handling

    All API functions return a matterfi_result_t structure containing:

    • error: Error code from the unified matterfi_error_t enum (defined above)
    • data: Generic data field containing the result (string, JSON, etc.)

    Example usage:

    matterfi_result_t result = matterfi_generate_seed();
if (result.error == MATTERFI_SUCCESS) {
    printf("Generated seed: %s\n", result.data);
    matterfi_string_free(result.data); // Free the data
} else {
    printf("Error: %d\n", result.error);
}
